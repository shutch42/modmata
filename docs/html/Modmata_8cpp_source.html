62/**
63 * @brief Read the command and args sent and execute the corresponding callback function,
64 * store the results of which in holding functions to be communicated with the host.
65 */
66void ModmataClass::processInput() {
67 // UNPACK COMMAND/FUNCTION CODE & NUMBER OF ARGS (ARGC)
68 uint16_t CMD_ARGC = mb.Hreg(0);
69 int cmd = highByte(CMD_ARGC);
70 int argc = lowByte(CMD_ARGC);
71
72 // Allocate space for argv to be transferred
73 uint8_t *argv = (uint8_t *)malloc(sizeof(uint8_t) * argc);
74
75 // Read Hregs into argv
76 for(int i = 0; i < argc; i++) {
77 uint16_t thisWord = mb.Hreg(i/2 + 1); // read Hreg()
78
79 // set argv[i] to the low or high byte of the current holding register depending the parity of the current index
80 argv[i] = (i % 2 == 0 ? highByte(thisWord) : lowByte(thisWord));
81 }
82
83 // EXECUTE CALLBACK FUNCTION
84 struct registers result = (callbackFunctions[cmd])(argc, argv);
85
86 // RESPOND WITH RESULT
87 for(int i = 0; i < result.count; i++) {
88 uint16_t thisWord = mb.Hreg(i/2 + 1); // Get the value of the current holding register
89 uint8_t curResult = result.value[i]; // Get the value of the current result
90
91 // Write half the holding register at a time, depending on whether
92 // it will be the high or low byte (determined by the parity of i)
93 mb.Hreg(i/2 + 1, (i % 2 == 0 ? makeWord(curResult, 0) : thisWord | curResult));
94 }
95
96 // Deallocate memory
97 free(argv);
98 if (result.value != nullptr) free(result.value);
99
100 // Save the number of result values, return to idle command
101 mb.Hreg(0, result.count);
102}

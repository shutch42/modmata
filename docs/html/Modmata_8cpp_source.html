80/**
81 * @brief Read the command and args sent and execute the corresponding callback function,
82 * store the results of which in holding functions to be communicated with the host.
83 */
84void ModmataClass::processInput() {
85 // Unpack all of the modbus registers as 8-bit values
86 byte cmd = mb.Hreg(0) >> 8; // 'Top' 8 bits of Hreg 0 indicate the command number
87 byte argc = mb.Hreg(0) & 0x00ff; // 'Bottom' 8 bits of Hreg 0 indicate the number of arguments for a command
88
89 size_t n_hregs = ceil(argc / 2); // Number of holding registers occupied by the command args
90 uint16_t hreg_array[ n_hregs ] = {}; // Temp. array to hold argument values
91 for (int i{0}; i < n_hregs; i++) {
92 hreg_array[i] = mb.Hreg(i+1); // Copy over values from Hregs to temp array
93 }
94
95 uint8_t * argv = (uint8_t *)calloc(argc, sizeof(byte)); // Argument array for the command
96 memcpy(argv, &hreg_array, argc); // Copy over values into persisting array
97
98 struct registers result = (callbackFunctions[cmd])(argc, argv); // Execute fn, collect result
99 n_hregs = ceil(result.count / 2); // Get number of holding registers needed to store/return the result
100 // This is not the same as the number of arguments used in a command!
101
102 uint16_t return_array[ n_hregs ] = {}; // Temp. array to store the collected values
103 memcpy(return_array, result.value, result.count); // Copy over values
104 for (int i{0}; i < n_hregs; i++) {
105 mb.Hreg(i+1, return_array[i]); // Update Hregs with result
106 }
107
108 free(argv);
109 free(result.value);
110
111 // Save the number of result values, return to idle command
112 mb.Hreg(0, result.count); // Sets the 'bottom' 8 bits of the 16-bit holding register (result.count is of type <uint8_t>)
113 // to # return vals, & 'top' 8 bits to 0's
114}

117/**
118 * Update modbus registers and check if a command has been received
119 * @remark Will return false unless there is a Command function code besides IDLE in Hreg 0
120 * _and_ there was data received since the last time mb.task() was called
121 * @return True or false
122*/
123bool ModmataClass::available() {
124 return mb.task() && (mb.Hreg(0) >> 8);
125}

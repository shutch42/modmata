26/**
27 * @brief Begin listening for a Modmata connection over serial
28 * @param baud Set the baud rate of the listening serial connection
29 * @remark This is configured to be used between a host computer and Arduino
30 * Leonardo using a USB connector (hardwired in the case of the LattePanda Delta 3)
31 * The Leonardo is different from most other Arduinos in the usage of the serial connection,
32 * so if you modify this code, please keep that in mind.

34void ModmataClass::begin(int baud) {

54 callbackFunctions[WIREWRITE] = &(functions::wireWrite);

79/**
80 * @brief Read the command and args sent and execute the corresponding callback function,
81 * store the results of which in holding functions to be communicated with the host.
82 */
83void ModmataClass::processInput() {
84 // Unpack all of the modbus registers as 8-bit values
85 byte cmd = mb.Hreg(0) >> 8; // 'Top' 8 bits of Hreg 0 indicate the command number
86 byte argc = mb.Hreg(0) & 0x00ff; // 'Bottom' 8 bits of Hreg 0 indicate the number of arguments for a command
87
88 size_t n_hregs = ceil(argc / 2); // Number of holding registers occupied by the command args
89 uint16_t hreg_array[ n_hregs ] = {}; // Temp. array to hold argument values
90 for (int i{0}; i < n_hregs; i++) {
91 hreg_array[i] = mb.Hreg(i+1); // Copy over values from Hregs to temp array
92 }
93
94 uint8_t * argv = (uint8_t *)calloc(argc, sizeof(byte)); // Argument array for the command
95 memcpy(argv, &hreg_array, argc); // Copy over values into persisting array
96
97 struct registers result = (callbackFunctions[cmd])(argc, argv); // Execute fn, collect result
98 n_hregs = ceil(result.count / 2); // Get number of holding registers needed to store/return the result
99 // This is not the same as the number of arguments used in a command!
100
101 uint16_t return_array[ n_hregs ] = {}; // Temp. array to store the collected values
102 memcpy(return_array, result.value, result.count); // Copy over values
103 for (int i{0}; i < n_hregs; i++) {
104 mb.Hreg(i+1, return_array[i]); // Update Hregs with result
105 }
106
107 free(argv);
108 free(result.value);
109
110 // Save the number of result values, return to idle command
111 mb.Hreg(0, result.count); // Sets the 'bottom' 8 bits of the 16-bit holding register (result.count is of type <uint8_t>)
112 // to # return vals, & 'top' 8 bits to 0's
113}

116/**
117 * Update modbus registers and check if a command has been received
118 * @remark Will return false unless there is a Command function code besides IDLE in Hreg 0
119 * _and_ there was data received since the last time mb.task() was called
120 * @return True or false
121*/
122bool ModmataClass::available() {
123 return mb.task() && (mb.Hreg(0) >> 8);

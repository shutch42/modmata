346/**
347 * @brief Read data from the connected I2C peripheral
348 *
349 * @param argc The number of arguments contained within the 'argv' array
350 * @param argv The arguments to use within the function
351 * @return result: a 'registers' struct holding the return values of the called function
352 */
353registers functions::wireRead(uint8_t argc, uint8_t *argv) {
354 registers result;
355 if(argc == 3) {
356 // Argument format: addr | reg | num_bytes
357 uint8_t addr = argv[0];
358 uint8_t reg = argv[1];
359 uint8_t num_bytes = argv[2];
360 result.value = malloc(sizeof(uint8_t) * num_bytes);
361
362 Wire.beginTransmission(addr);
363 Wire.write(reg);
364 Wire.endTransmission();
365
366 Wire.requestFrom(addr, num_bytes);
367 if(Wire.available() == num_bytes) {
368 result.count = num_bytes;
369 for(int i = 0; i < num_bytes; i++) {
370 result.value[i] = Wire.read();
371 }
372 }
373 else {
374 result.count = 0;
375 }
376 }
377 else {
378 result.count = 0;
379 }
380
381 return result;
382}

428/**
429 * @brief Exchange data over the SPI connection (Read + Write)
430 *
431 * @param argc The number of arguments contained within the 'argv' array
432 * @param argv The arguments to use within the function
433 * @return result: a 'registers' struct holding the return values of the called function
434 */
435registers functions::spiTransferBuf(uint8_t argc, uint8_t *argv) {
436 // Register format:
437 // | CMD/ARGC | Bytes |
438 // |<-1 word->|<-31 words / 62 bytes->|
439
440 registers result;
441
442 if (argc > 1) {
443 uint8_t CS_pin = argv[0];
444
445 result.count = argc - 1;
446 result.value = malloc(sizeof(uint8_t) * (argc - 1));
447
448 SPI.beginTransaction(SPISettings(settings.speed, settings.order, settings.mode));
449 adigitalWrite((uint8_t)CS_pin, (uint8_t)LOW);
450 for(int i = 1; i < argc; i++) {
451 result.value[i-1] = SPI.transfer(argv[i]);
452 }
453 adigitalWrite((uint8_t)CS_pin, (uint8_t)HIGH);
454 SPI.endTransaction();
455 }
456 else {
457 result.count = 0;
458 }
459
460 return result;
461}
